using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class ropeSwing : MonoBehaviour {

    void Start() {
      var hinge = GetComponent<HingeJoint2D>(); // access the 2d hinge joint from the object the script is attached to
      var motor = hinge.motor; // makes hinge joint motor easier to access
      hinge.useMotor = true; // actives hinge joint motor
      motor.motorSpeed = 0; // sets a speed for the motor
      motor.maxMotorTorque = 1000; // how much horse power the motor has
      hinge.useLimits = false; // hinge joint has a full range of motion
      hinge.motor = motor; // var motor = hinge.motor; reversed (script will not work without this line)
    }
}
